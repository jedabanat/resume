% Sets the page style to empty, with no headers or footers.
\pagestyle{empty}

% Suppress bad box warnings.
\hbadness=10000

% Sets the space between columns in tables to 0em, reducing extra space.
\setlength{\tabcolsep}{0em}

% Allow unlimited stretching of spaces to avoid overfull lines in emergencies.
\emergencystretch=\maxdimen

% Sets the paragraph indentation to zero, eliminating the space at the beginning of each paragraph.
\setlength{\parindent}{0pt}
